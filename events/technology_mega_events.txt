event = {
	id = megatech.1
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		NOT = { has_global_flag = acot_activated }
		end_game_years_passed >= 100
	}
	
	immediate = {
		random_country = {
			limit = {
				NOT = { has_country_flag = researching_dark_matter_power }
				NOT = { has_technology = tech_dark_matter_power_core }
				NOT = { has_tech_option = tech_dark_matter_power_core }
				has_technology = tech_dark_matter_plant
				has_technology = tech_zero_point_power
				has_technology = tech_power_plant_4
				has_technology = tech_hyper_drive_3
				has_technology = tech_physics_3
				has_technology = tech_basic_science_lab_3
				has_technology = tech_galactic_administration
				has_technology = tech_engineering_3
				has_technology = tech_battleships
				has_technology = tech_starbase_5
			}
			set_country_flag = researching_dark_matter_power
			country_event = { id = megatech.2 days = 30 random = 300 }
		}
	}
}

			}

		name = megatech.2.a
		capital_scope = {
			enable_special_project = {
				name = "DARK_MATTER_SECRETS"

				owner = root
			}
		}
	}
	option = {
		name = megatech.2.b
		trigger = {
			exists = event_target:the_curators
		}
		allow = {
			custom_tooltip = {
				text = megatech.2.b.trust

					trust = {
						who = root
						value >= 100
					}
				}
			}
		}

	}
}
